       class CreateBackupUseCase extends UseCase<BackupMetadata, CreateBackupParams> {
         final BackupWriteRepository _backupRepo;
         final AuthRepository _authRepo;

         CreateBackupUseCase(this._backupRepo, this._authRepo);

         @override
         Future<Result<BackupMetadata>> call(CreateBackupParams params) async {
           // Ensure authenticated first
           final authResult = await _authRepo.getSignedInUser();
           if (authResult.isFailure || authResult.data == null) {
             final signInResult = await _authRepo.signIn();
             if (signInResult.isFailure) return Result.failure(signInResult.failure!);
           }

           // Create backup with progress callback
           return _backupRepo.createBackup(onProgress: params.onProgress);
         }
       }

       class CreateBackupParams {
         final void Function(double progress)? onProgress;
         CreateBackupParams({this.onProgress});
       }
       ```

       @riverpod
       class BackupController extends _$BackupController {
         @override
         BackupProgress build() => const BackupProgressIdle();

         Future<void> createBackup() async {
           state = const BackupProgressSerializing();
           final useCase = ref.read(createBackupUseCaseProvider);
           final result = await useCase(CreateBackupParams(
             onProgress: (progress) {
               state = BackupProgressUploading(progress);
             },
           ));
           if (result.isSuccess) {
             state = BackupProgressCompleted(result.data!);
           } else {
             state = BackupProgressFailed(
               ErrorMessageMapper.getUserMessage(result.failure),
             );
           }
         }

         void reset() {
           state = const BackupProgressIdle();
         }
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| App → Google Drive | Backup data uploaded to Google servers in app-only folder (drive.appdata) |
| Local DB → Serializer | All local data read for serialization; no data leaves device until explicit user action |
| Serializer → Upload | Backup bytes in memory temporarily during upload |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-06 | I (Info Disclosure) | Backup data at rest on Drive | accept | drive.appdata scope means only this app can access; no user files accessible. No PII beyond what user entered. |
| T-04-07 | T (Tampering) | Backup file integrity | mitigate | Schema version check rejects incompatible/corrupted backups |
| T-04-08 | D (Denial) | Upload failure mid-transfer | mitigate | Transactional: if upload fails, no partial backup on Drive. User can retry. |
</threat_model>
