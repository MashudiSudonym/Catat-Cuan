abstract class SavingsGoalReadRepository {
  Future<Result<List<SavingsGoalEntity>>> getGoals({required String status});
  Future<Result<SavingsGoalEntity>> getGoalById(int id);
  Future<Result<List<SavingsGoalEntity>>> getActiveGoals();
}

abstract class SavingsGoalWriteRepository {
  Future<Result<SavingsGoalEntity>> createGoal({...});
  Future<Result<SavingsGoalEntity>> updateGoal(int id, {...});
  Future<Result<void>> softDeleteGoal(int id);
}

abstract class SavingsGoalContributionRepository {
  Future<Result<GoalContributionEntity>> addContribution(int goalId, double amount, String? note, DateTime date);
  Future<Result<GoalContributionEntity>> withdrawFromGoal(int goalId, double amount, String? note, DateTime date);
  Future<Result<List<GoalContributionEntity>>> getContributionsForGoal(int goalId);
}

abstract class SavingsGoalQueryRepository {
  Future<Result<List<SavingsGoalWithProgressEntity>>> getGoalsWithProgress();
  Future<Result<SavingsGoalWithProgressEntity>> getGoalWithProgressById(int id);
  Future<Result<({double totalSaved, double totalTarget, double percentage})>> getOverallProgress();
}
